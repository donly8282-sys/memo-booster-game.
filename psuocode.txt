// Main Pages
// Page 1: Home Page
FUNCTION homePage()
    DISPLAY "Welcome to Dinar Connection"
    DISPLAY "Start Game" BUTTON
    DISPLAY "How to Play" BUTTON
    DISPLAY "Leaderboard" BUTTON
    DISPLAY "Settings" BUTTON
    IF "Start Game" BUTTON CLICKED THEN
        CALL gamePage()
    IF "How to Play" BUTTON CLICKED THEN
        CALL tutorialPage()
    IF "Leaderboard" BUTTON CLICKED THEN
        CALL leaderboardPage()
    IF "Settings" BUTTON CLICKED THEN
        CALL settingsPage()
END FUNCTION

// Page 2: Game Page
